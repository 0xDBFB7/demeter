Linear Combination Fitting

LCF 

    

  1. #!/usr/bin/perl

  2. use Demeter qw(:ui=screen :plotwith=gnuplot);

  3. 

  4. my $prj = Demeter::Data::Prj -> new(file=>'../../cyanobacteria.prj');

  5. my $lcf = Demeter::LCF -> new(space=>'nor', plot_difference=>1, plot_components=>1);

  6. 

  7. $prj -> set_mode('screen' => 1);

  8. 

  9. my $data = $prj->record(4);

  10. my ($metal, $chloride, $sulfide) = $prj->records(9, 11, 15);

  11. 

  12. $lcf->data($data);

  13. $lcf->add_many($metal, $chloride, $sulfide);

  14. 

  15. $lcf->xmin($data->bkg_e0-20);

  16. $lcf->xmax($data->bkg_e0+60);

  17. $lcf->po->set(emin=>-30, emax=>80);

  18. 

  19. $lcf -> fit -> plot -> save('lcf_fit_result.dat');

  20. $lcf->pause;

  21. 

  22. $lcf->clean;
